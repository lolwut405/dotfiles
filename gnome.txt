mkdir -p ~/.mozilla/native-messaging-hosts
chmod +x ff2mpv.py
sed -i 's:/home/william/scripts/ff2mpv:/home/blah/.mozilla/native-messaging-hosts/ff2mpv.py:' ff2mpv.json
cp ff2mpv.py ff2mpv.json ~/.mozilla/native-messaging-hosts/
cd ..; rm -rf ff2mpv
mkdir ~/.config/mpv
echo 'keep-open=yes' >> ~/.config/mpv/mpv.conf
echo 'x11-bypass-compositor=no' >> ~/.config/mpv/mpv.conf

# Disable mouse acceleration
cat <<EOF >> /etc/X11/xorg.conf.d/50-mouse-acceleration.conf
Section "InputClass"
	Identifier "My Mouse"
	Driver "libinput"
	MatchIsPointer "yes"
	Option "AccelProfile" "flat"
	Option "AccelSpeed" "0"
EndSection
EOF
